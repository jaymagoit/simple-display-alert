Simple Display Alert

A lightweight, customizable alert system for modern web applications

Installation

npm install simple-display-alert

Basic Usage

import { 
    displayAlert, 
    displayAlertSuccess, 
    displayAlertError, 
    displayAlertWarning, 
    displayAlertInfo,
    displayAlertLoader
} from 'simple-display-alert';

// Using specific alert functions
displayAlertSuccess('Operation completed successfully!');
displayAlertError('An error occurred!');
displayAlertWarning('Please check your input');
displayAlertInfo('This is an informational message');
displayAlertLoader('Loading...');

// Using specific alert functions with options
displayAlertSuccess('Success with options', { position: 'top-right', fadeDirection: 'left' });
displayAlertError('Error with options', { timeout: 0, multiple: true });
displayAlertWarning('Warning with options', { position: 'bottom-left', fadeDirection: 'bottom' });
displayAlertInfo('Info with options', { timeout: 3000, multiple: true });
displayAlertLoader('Loading with options', { position: 'top-right', timeout: 2000 });

// With options
displayAlertSuccess({
    message: 'Operation completed successfully!',
    timeout: 5000,
    position: 'top-right',
    fadeDirection: 'left'
});

// Using generic displayAlert
displayAlert({
    variant: 'success', // 'success' | 'error' | 'warning' | 'info' | 'white' | 'loader'
    message: 'Your message here',
    timeout: 5000,      // 0 for no auto-hide
    position: 'bottom-right', // 'top-left' | 'top-right' | 'bottom-left' | 'bottom-right'
    fadeDirection: 'left', // 'left' | 'right' | 'top' | 'bottom'
    icon: 'success',    // 'success' | 'danger' | 'warning' | 'info' | 'loader'
    closeBtn: true,     // show/hide close button
    multiple: false     // allow multiple alerts
});

// Hiding alerts programmatically
// Method 1: Store the alert reference
const alert = displayAlert({
    variant: 'info',
    message: 'This alert can be hidden programmatically',
    timeout: 0
});
// Later, hide it using:
alert.remove();

// Method 2: Remove alerts by position
// Import the removal function
import { removeResponseAlert } from 'simple-display-alert';

// Remove all alerts in all positions
removeResponseAlert(); // Removes all alerts from the page
